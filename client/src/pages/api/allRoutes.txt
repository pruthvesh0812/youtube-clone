#db schema

postgres

table - user - pk 
      - userid
      - email
      - password

table - channel - ref userid
      - channelId
      - channelName
      - channelSubs
      - channelProfilePhoto
      - userId(reference from user table)

table - videos
      - videoId
      - channelId (reference from channel table)
      - videoTitle
      - videoDescription
      - numberOfLikes
      - numberOfViews
      - publishedDate
      - thumbnail photo
